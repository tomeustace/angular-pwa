‍ Add PWA via CLI️

Run command: ng add @angular/pwa

  • Creates web manifest file
  • Creates ngsw-config.json (ServiceWorker config)
  • Creates icon files
  • Updates angular.json to reference ngsw-config.json, set ServiceWorker true
  • Updates index.html to link webmanifest and include theme color
  • Updates app.module.ts to register ServiceWorker script ngsw-worker.js (generated)
 ServiceWorker Config - assetGroups

assetGroups defines what files get cached by the ServiceWorker.

{{ '{' }}
  "$schema": "./node_modules/@angular/service-worker/config/schema.json",
  "index": "/index.html",
  "assetGroups": [
    {{ '{' }}
      "name": "app",
      "installMode": "prefetch",
      "resources": {{ '{' }}
        "files": [ "/favicon.ico", "/index.html", "/manifest.webmanifest", "/*.css", "/*.js" ]
      }
    },
    {{ '{' }}
      "name": "assets",
      "installMode": "lazy",
      "updateMode": "prefetch",
      "resources": {{ '{' }}
        "files": [
          "/assets/**",
          "/*.(svg|cur|jpg|jpeg|png|apng|webp|avif|gif|otf|ttf|woff|woff2)"
        ]
      {{ '}' }}
  {{ '}' }}
  ]
{{ '}' }}
 ServiceWorker Config - dataGroups

dataGroups defines policies for how API requests are cached.

{{ '{' }}
  "$schema": "./node_modules/@angular/service-worker/config/schema.json",
  "index": "/index.html",
  "dataGroups": [
    {{ '{' }}
      "name": "users",
      "urls": [ "/api/users/**" ],
      "cacheConfig": {{ '{' }}
        "strategy": "performance", "maxSize": 100, "maxAge": "5m"
      {{ '}' }}
    {{ '}' }},
    {{ '{' }}
      "name": "products",
      "urls": [ "/api/products/**" ],
      "cacheConfig": {{ '{' }}
        "strategy": "freshness", "maxSize": 50, "maxAge": "15s"
      {{ '}' }}
    {{ '}' }}
  ]
{{ '}' }}
 Web Manifest

  "name": "angular-pwa",

  "scope": "./",
  "start_url": "./",
  "icons": [...]
{{ '}' }}
